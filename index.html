							</script>

In Node.js:


							$ npm install @tensorflow/tfjs
						

							import * as tf from '@tensorflow/tfjs';

Tensors

Creating a tensor

tf.tensor(values, shape?, dtype?)


							values (TypedArray|Array)
							shape (number[])
							dtype ('float32'|'int32'|'bool'|'complex64')

Variables

tf.variable()


							const x = tf.variable(tf.tensor([1, 2, 3]));
							x.assign(tf.tensor([4, 5, 6]));
							
							x.print();

Operations

tf.add()


							const a = tf.tensor1d([1, 2, 3, 4]);
							const b = tf.tensor1d([10, 20, 30, 40]);
							
							a.add(b).print();  // or tf.add(a, b)
							// [11, 22, 33, 44]

tf.matmul()


							const a = tf.tensor2d([1, 2], [1, 2]);
							const b = tf.tensor2d([1, 2, 3, 4], [2, 2]);
							
							a.matMul(b).print();  // or tf.matMul(a, b)

More operations on the documentation page

Memory Management

  1. tf.dispose() — Disposes any tf.Tensors found within the provided object.
  2. tf.tidy() — Using this method helps avoid memory leaks.

Creating from scratch and training simple model

Code example


							1	// 1. Define a model for linear regression
							2	const model = tf.sequential();
							3	model.add(tf.layers.dense({units: 1, inputShape: [1]}));
							4	
							5	// 2. Prepare the model for training
							6	model.compile({loss: 'meanSquaredError', optimizer: 'sgd'});
							7
							8	// 3. Generate some synthetic data for training
							9	const xs = tf.tensor2d([1, 2, 3, 4], [4, 1]);
							10	const ys = tf.tensor2d([1, 3, 5, 7], [4, 1]);
							11
							12	// 4. Train the model using the data
							13	model.fit(xs, ys, {epochs: 10}).then(() => {
							14	  // 5. Use the model
							15	  model.predict(tf.tensor2d([5], [1, 1])).print();
							16	});
